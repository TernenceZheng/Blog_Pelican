for (let group of dv.pages().where(p => p.category != null).groupBy(p => p.category)) {
    dv.header(2, group.key);
    dv.table(["檔案名稱", "建立日期"],
        group.rows
			.where(w => w.category != null)
			.where(w => w.status != 'draft')
            .sort(k => k.date, 'desc')
            .map(k => [k.file.link, k.date ]))
}
